app.get('/api/today-slots', async (req, res) => {
  const today = new Date().toISOString().split('T')[0]; // Ambil tanggal hari ini

  const slots = await db.query(`
    SELECT 
      time_slot, 
      max_quota, 
      current_count,
      (current_count * 100 / max_quota) AS percentage
    FROM therapy_slots
    WHERE date = $1
  `, [today]);

  res.json(slots);
});

import { useEffect, useState } from 'react';

function SlotTracker() {
  const [slots, setSlots] = useState([]);

  useEffect(() => {
    fetch('/api/today-slots')
      .then(response => response.json())
      .then(data => setSlots(data));
  }, []);

  return (
    <div className="slot-tracker">
      {slots.length > 0 ? (
        <table>
          <thead>
            <tr>
              <th>Waktu</th>
              <th>Kuota</th>
              <th>Sudah Terisi</th>
              <th>Persentase</th>
            </tr>
          </thead>
          <tbody>
            {slots.map(slot => (
              <tr key={slot.time_slot}>
                <td>{slot.time_slot}</td>
                <td>{slot.max_quota}</td>
                <td>{slot.current_count}</td>
                <td>
                  <progress value={slot.percentage} max="100"></progress>
                  {slot.percentage.toFixed(0)}%
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      ) : (
        <div className="empty-state">
          <img src="/calendar-icon.svg" alt="Kalender" />
          <p>Belum Ada Janji Hari Ini</p>
          <small>
            Kunjungi halaman Therapy Slots untuk menambahkan slot baru.
          </small>
        </div>
      )}
    </div>
  );
}

export default SlotTracker;

4. Fitur Tambahan
Filter Berdasarkan Tanggal:
Tambahkan dropdown untuk memilih tanggal lain (opsional).
Notifikasi Ketika Slot Penuh:
Ubah warna latar menjadi merah jika current_count >= max_quota.
Realtime Update:
Gunakan WebSocket untuk memperbarui data tanpa refresh halaman.
5. Catatan Penting
Pastikan kolom date di tabel therapy_slots menggunakan format ISO (YYYY-MM-DD).
Uji coba dengan skenario:
Slot kosong hari ini.
Slot penuh dan belum penuh.
Jam server sesuai zona waktu lokal.
Dengan fitur ini, admin bisa melihat status pengisian slot secara instan dari dashboard! 📊
